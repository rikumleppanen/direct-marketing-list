5   import java.io.File;

7   import java.io.PrintWriter;
8   import java.util.logging.Level;
9   import java.util.logging.Logger;

11  public class FileWriter {
12  
13      private int countRows;
14  
15      public FileWriter() {
16          this.countRows = 0;
17      }
18  
19      public void write(String row, MarketingList list) {
20          if (list.getRows() == 0) {
21              System.out.println("Marketing List is empty");
22          } else {
23              try (PrintWriter write = new PrintWriter(new File(row))) {
24                  for (Consent one : list.getListOfConsents()) {
25                      write.println(one.getRow());
26                      countRows++;
27                  }
28                  write.flush();
29                  write.close();
30  
31              } catch (FileNotFoundException ex) {
32                  Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
33              }
34          }
35      }
36  
37      public int getRowCount() {
38          return this.countRows;
39      }
40  }
