206     private void addContactListButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_addContactListButtonActionPerformed
207         if (evt.getSource() == addContactListButton && jTextField1.getText().endsWith(".txt")) {
208             InputStream is = getClass().getClassLoader().getResourceAsStream(jTextField1.getText());
209             FileReader just = oper.fileReaderInputStream(is);
210             long startTime = System.currentTimeMillis();
211             oper.matchAContactListToCustomers(just, contactListName.getText());
212             long endTime = System.currentTimeMillis();
213             infoArea1.setText(oper.printTime(startTime, endTime) + "\n" + "There is " + oper.getContactCountNumber() + " rows of contact data to be put on the list " + contactListName.getText() + ".");
214             infoAreaStatistics.setText("There are " + oper.getCustomers().numberOfCustomers() + " customers in the system. \n Emails: " + oper.getCustomers().getNumberOfEmails() + "\n Phone numbers: " + oper.getCustomers().getNumberOfPhoneNumbers());
215         }
216     }//GEN-LAST:event_addContactListButtonActionPerformed

232     private void setTextAndType(javax.swing.JTextArea two, javax.swing.JTextArea three, fi.marketing.list.logic.Type tp) {

238     private void openAFileChooser() {
239         FileFilter ft = new FileNameExtensionFilter("Text Files", "txt");
240         db.addChoosableFileFilter(ft);
241         int returnVal = db.showOpenDialog(this);     
242 
243         if (returnVal == javax.swing.JFileChooser.APPROVE_OPTION) {
244             java.io.File file = db.getSelectedFile();
245             if (file.isFile()) {
246                 String file_name = file.toString();
247                 //InputStream is = getClass().getClassLoader().getResourceAsStream(file_name);
248                 //FileReader just = oper.fileReaderInputStream(is);
249                 String name = FilenameUtils.getName(file_name);
250                 jTextField1.setText(name);
251             }
252         }
