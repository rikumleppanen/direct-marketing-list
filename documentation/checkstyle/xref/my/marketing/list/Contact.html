24          java.util.regex.Pattern p = java.util.regex.Pattern.compile(ePattern);
25          java.util.regex.Matcher m = p.matcher(this.row);

41              setRow(this.row.replace("-", ""));
42          }
43          if (this.row.contains("(")) {
44              setRow(this.row.replace("(", ""));
45          }
46          if (this.row.contains(")")) {
47              setRow(this.row.replace(")", ""));
48          }
49          if (this.row.contains("+3580")) {
50              setRow(this.row.replace("+3580", ""));
51          }
52          if (this.row.contains("+358")) {
53              setRow(this.row.replace("+358", "0"));
54          }
55          if (this.row.startsWith("3580") == true) {
56              setRow(this.row.replace("358", ""));
57          }
58          if (this.row.startsWith("358") == true) {
59              setRow(this.row.replace("358", "0"));
