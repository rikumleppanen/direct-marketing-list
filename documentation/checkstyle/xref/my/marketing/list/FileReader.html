3   import java.io.File;

6   import java.util.ArrayList;

10  import java.util.logging.Logger;

12  public class FileReader {
13  
14      private List<Contact> list;
15      private int countRows;
16  
17      public FileReader() {
18          this.list = new ArrayList<>();
19          this.countRows = 0;
20      }
21  
22      public void read(String file) {
23          try (Scanner reader = new Scanner(new File(file), "UTF-8")) {
24              while (reader.hasNextLine()) {
25                  String row = reader.nextLine();
26                  String[] items = row.split("\\|");
27                  int id = Integer.valueOf(items[1]);
28                  list.add(new Contact(items[0], id));
29                  countRows++;
30              }
31          } catch (FileNotFoundException ex) {
32              Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
33          }
34      }
35  
36      public List getList() {
37          return this.list;
38      }
39  
40      public int getNumberOfRows() {
41          return this.countRows;
42      }
43  
44  }
